## Cluster

<ResponseField name="cluster_arn" type="string">
  The ARN of the cluster.
</ResponseField>

<ResponseField name="cluster_id" type="string">
  The ID of the cluster.
</ResponseField>

<ResponseField name="cluster_core_instance_group_id" type="string">
  Core node type Instance Group ID, if using Instance Group for this node type.
</ResponseField>

<ResponseField name="cluster_master_instance_group_id" type="string">
  Master node type Instance Group ID, if using Instance Group for this node type.
</ResponseField>

<ResponseField name="cluster_master_public_dns" type="string">
  The DNS name of the master node. If the cluster is on a private subnet, this is the private DNS name. On a public subnet, this is the public DNS name.
</ResponseField>
